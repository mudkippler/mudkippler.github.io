namespace YourApp.Extensions

    public static class TomSelectExtensions
    {
        public static MvcHtmlString FoodTypesDropdownFor<TModel, TProperty>(
            this HtmlHelper<TModel> htmlHelper,
            Expression<Func<TModel, TProperty>> expression,
            IEnumerable<FoodType> foodTypes,
            object htmlAttributes = null)
        {
            var name = ExpressionHelper.GetExpressionText(expression);
            var fullName = htmlHelper.ViewContext.ViewData.TemplateInfo.GetFullHtmlFieldName(name);
            var metadata = ModelMetadata.FromLambdaExpression(expression, htmlHelper.ViewData);
            var value = metadata.Model;
            
            var items = foodTypes.Select(ft => new SelectListItem 
            { 
                Value = ft.Id.ToString(), 
                Text = ft.Name 
            });
            
            var viewData = new ViewDataDictionary(htmlHelper.ViewData)
            {
                ["Name"] = fullName,
                ["SelectedValue"] = value,
                ["Items"] = items,
                ["HtmlAttributes"] = htmlAttributes
            };
            
            return htmlHelper.Partial("_FoodTypesDropdown", viewData);
        }

        public static MvcHtmlString UsersDropdownFor<TModel, TProperty>(
            this HtmlHelper<TModel> htmlHelper,
            Expression<Func<TModel, TProperty>> expression,
            IEnumerable<User> users,
            object htmlAttributes = null)
        {
            var name = ExpressionHelper.GetExpressionText(expression);
            var fullName = htmlHelper.ViewContext.ViewData.TemplateInfo.GetFullHtmlFieldName(name);
            var metadata = ModelMetadata.FromLambdaExpression(expression, htmlHelper.ViewData);
            var value = metadata.Model;
            
            var items = users.Select(u => new SelectListItem 
            { 
                Value = u.Id.ToString(), 
                Text = $"{u.FirstName} {u.LastName}" 
            });
            
            var viewData = new ViewDataDictionary(htmlHelper.ViewData)
            {
                ["Name"] = fullName,
                ["SelectedValue"] = value,
                ["Items"] = items,
                ["HtmlAttributes"] = htmlAttributes
            };
            
            return htmlHelper.Partial("_UsersDropdown", viewData);
        }

        public static MvcHtmlString TagsDropdownFor<TModel, TProperty>(
            this HtmlHelper<TModel> htmlHelper,
            Expression<Func<TModel, TProperty>> expression,
            IEnumerable<Tag> tags,
            object htmlAttributes = null)
        {
            var name = ExpressionHelper.GetExpressionText(expression);
            var fullName = htmlHelper.ViewContext.ViewData.TemplateInfo.GetFullHtmlFieldName(name);
            var metadata = ModelMetadata.FromLambdaExpression(expression, htmlHelper.ViewData);
            var value = metadata.Model;
            
            var items = tags.Select(t => new SelectListItem 
            { 
                Value = t.Id.ToString(), 
                Text = t.Name 
            });
            
            var viewData = new ViewDataDictionary(htmlHelper.ViewData)
            {
                ["Name"] = fullName,
                ["SelectedValue"] = value,
                ["Items"] = items,
                ["HtmlAttributes"] = htmlAttributes
            };
            
            return htmlHelper.Partial("_TagsDropdown", viewData);
        }
    }

    var items = (IEnumerable<SelectListItem>)ViewBag.Items;
    var htmlAttributes = ViewBag.HtmlAttributes;

    var attributes = htmlAttributes != null 
        ? HtmlHelper.AnonymousObjectToHtmlAttributes(htmlAttributes)

        ? "form-control " + attributes["class"]
        : "form-control";

<script>
(function() {
    var dropdown = new TomSelect('#@dropdownId', {
        plugins: ['clear_button'],
        searchField: ['text'],
        placeholder: 'Type to search users...',
        preload: false,
        load: function(query, callback) {
            if (!query.length) return callback();
            
            // Make AJAX call to search users
            $.ajax({
                url: '@Url.Action("SearchUsers", "Users")',
                type: 'GET',
                dataType: 'json',
                data: { q: query },
                success: function(data) {
                    callback(data.map(function(user) {
                        return {
                            value: user.id,
                            text: user.firstName + ' ' + user.lastName
                        };
                    }));
                },
                error: function() {
                    callback();
                }
            });
        },
        onChange: function(value) {
            // Handle change - call global handler if available
            if (window.userHandlers && window.userHandlers.onChange) {
                window.userHandlers.onChange(value, '@dropdownId');
            }
        }
    });
})();

@{
    var dropdownId = "tags-" + Guid.NewGuid().ToString("N")[..8];
    var name = ViewBag.Name;
    var selectedValue = ViewBag.SelectedValue;
    var items = (IEnumerable<SelectListItem>)ViewBag.Items;
    var htmlAttributes = ViewBag.HtmlAttributes;
    
    // Convert anonymous object to dictionary for easier handling
    var attributes = htmlAttributes != null 
        ? HtmlHelper.AnonymousObjectToHtmlAttributes(htmlAttributes)
        : new Dictionary<string, object>();
    
    // Ensure we have a class attribute, merging with any provided
    var cssClass = attributes.ContainsKey("class") 
        ? "form-control " + attributes["class"]
        : "form-control";
    attributes["class"] = cssClass;
    
    // Handle multi-select values (could be array or comma-separated string)
    var selectedValues = new List<string>();
    if (selectedValue != null)
    {
        if (selectedValue is IEnumerable<object> enumerable)
        {
            selectedValues = enumerable.Select(v => v.ToString()).ToList();
        }
        else
        {
            selectedValues = selectedValue.ToString().Split(',').Select(v => v.Trim()).ToList();
        }
    }
}

<select id="@dropdownId" name="@name" multiple @Html.Raw(string.Join(" ", attributes.Select(a => $"{a.Key}=\"{Html.Encode(a.Value)}\"")))>
    @foreach(var item in items)
    {
        <option value="@item.Value" @(selectedValues.Contains(item.Value) ? "selected" : "")>
            @item.Text
        </option>
    }
</select>

<script>
(function() {
    var dropdown = new TomSelect('#@dropdownId', {
        plugins: ['remove_button'],
        searchField: ['text'],
        placeholder: 'Select tags...',
        maxItems: null, // Allow unlimited selections
        create: false,  // Don't allow creating new tags
        onChange: function(values) {
            // Handle change - call global handler if available
            if (window.tagHandlers && window.tagHandlers.onChange) {
                window.tagHandlers.onChange(values, '@dropdownId');
            }
        },
        onItemAdd: function(value, item) {
            // Handle item add - call global handler if available
            if (window.tagHandlers && window.tagHandlers.onItemAdd) {
                window.tagHandlers.onItemAdd(value, item, '@dropdownId');
            }
        },
        onItemRemove: function(value) {
            // Handle item remove - call global handler if available
            if (window.tagHandlers && window.tagHandlers.onItemRemove) {
                window.tagHandlers.onItemRemove(value, '@dropdownId');
            }
        }
    });
})();
</script>

public JsonResult SearchUsers(string q)
{
    var users = _userService.SearchUsers(q)
        .Select(u => new { 
            id = u.Id, 
            firstName = u.FirstName, 
            lastName = u.LastName 
        })
        .Take(10)
        .ToList();
    
    return Json(users, JsonRequestBehavior.AllowGet);
}
